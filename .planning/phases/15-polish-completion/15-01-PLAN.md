private getCurrentAct(): 1 | 2 | 3 {
  try {
    const raw = localStorage.getItem('pq:save:v2');
    if (!raw) return 1;
    const save = JSON.parse(raw);
    const act = save?.actProgress;
    if (act === 2 || act === 3) return act;
  } catch { /* ignore */ }
  return 1;
}
