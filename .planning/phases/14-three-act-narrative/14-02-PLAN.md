Implement `useGameState` hook:

```typescript
import { useState, useCallback, useEffect, useRef } from 'react';
import { ActState, DecisionState, DEFAULT_ACT_STATE, DEFAULT_DECISIONS, ACT_MUSIC_MAP, ACT3_MUSIC_BASE_VOLUME } from '../types/narrative';
import { eventBridge, BRIDGE_EVENTS } from '../phaser/EventBridge';

const LS_ACT_KEY = 'pq:act:state';
const LS_DEC_KEY = 'pq:decisions';

function loadActState(): ActState {
  try {
    const s = localStorage.getItem(LS_ACT_KEY);
    return s ? { ...DEFAULT_ACT_STATE, ...JSON.parse(s) } : { ...DEFAULT_ACT_STATE };
  } catch { return { ...DEFAULT_ACT_STATE }; }
}

function loadDecisions(): DecisionState {
  try {
    const s = localStorage.getItem(LS_DEC_KEY);
    return s ? { ...DEFAULT_DECISIONS, ...JSON.parse(s) } : { ...DEFAULT_DECISIONS };
  } catch { return { ...DEFAULT_DECISIONS }; }
}

export function useGameState() {
  const [actState, setActStateInternal] = useState<ActState>(loadActState);
  const [decisions, setDecisionsInternal] = useState<DecisionState>(loadDecisions);

  // Keep refs in sync for safe reading in callbacks without stale closures
  const actRef = useRef(actState);
  const decisionsRef = useRef(decisions);
  useEffect(() => { actRef.current = actState; }, [actState]);
  useEffect(() => { decisionsRef.current = decisions; }, [decisions]);

  const setActState = useCallback((next: ActState) => {
    setActStateInternal(next);
    localStorage.setItem(LS_ACT_KEY, JSON.stringify(next));
  }, []);

  const setDecisions = useCallback((next: DecisionState) => {
    setDecisionsInternal(next);
    localStorage.setItem(LS_DEC_KEY, JSON.stringify(next));
  }, []);

  const checkActAdvance = useCallback((completedRooms: string[]) => {
    const act = actRef.current;

    if (act.current === 1 && !act.act1Complete) {
      if (completedRooms.includes('reception') && completedRooms.includes('break_room')) {
        const next: ActState = {
          ...act,
          current: 2,
          act1Complete: true,
          musicTrack: 'music_exploration',
        };
        setActState(next);
        eventBridge.emit(BRIDGE_EVENTS.ACT_ADVANCE, { newAct: 2, track: 'music_exploration' });
        return;
      }
    }

    if (act.current === 2 && !act.act2Complete) {
      if (completedRooms.includes('lab') && completedRooms.includes('records_room')) {
        const next: ActState = {
          ...act,
          current: 3,
          act2Complete: true,
          musicTrack: 'music_breach',
        };
        setActState(next);
        // Act 3 breach music plays quieter during RPG exploration (polish debt — Phase 15)
        eventBridge.emit(BRIDGE_EVENTS.ACT_ADVANCE, {
          newAct: 3,
          track: 'music_breach',
          baseVolume: ACT3_MUSIC_BASE_VOLUME,  // 0.15 vs default 0.25
        });
        return;
      }
    }
  }, [setActState]);

  return { actState, decisions, setActState, setDecisions, checkActAdvance };
}
```
