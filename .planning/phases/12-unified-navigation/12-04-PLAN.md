**Key function — computeDoorStates:**
```typescript
function computeDoorStates(
  room: RoomWithDoors,
  completedRooms: string[]
): Record<string, 'locked' | 'available' | 'completed'> {
  const states: Record<string, 'locked' | 'available' | 'completed'> = {};
  if (!room.doors) return states;
  for (const door of room.doors) {
    const target = door.targetRoomId;
    if (!isDepartmentAccessible(target, completedRooms)) {
      states[door.id] = 'locked';
    } else if (completedRooms.includes(target)) {
      states[door.id] = 'completed';
    } else {
      states[door.id] = 'available';
    }
  }
  return states;
}
```

**Key handler — EXPLORATION_EXIT_ROOM:**
```typescript
const handleExitRoom = useCallback(({ targetRoomId, fromDoorId }: { targetRoomId: string; fromDoorId: string }) => {
  if (!isDepartmentAccessible(targetRoomId, gameState.state.completedRooms)) {
    // Door is locked — send feedback to scene
    eventBridge.emit(BRIDGE_EVENTS.REACT_DOOR_LOCKED);
    return;
  }

  // Find the target room data
  const nextRoom = rooms.find(r => r.id === targetRoomId);
  if (!nextRoom) return;

  // Find the spawn door on the other side (the door that leads back to current room)
  const spawnDoor = nextRoom.doors?.find((d: DoorData) => d.targetRoomId === gameState.state.currentRoomId);

  // Update state
  gameState.setCurrentRoom(targetRoomId);

  // Compute door states for the new room
  const doorStates = computeDoorStates(nextRoom, gameState.state.completedRooms);

  // Send new room to ExplorationScene
  eventBridge.emit(BRIDGE_EVENTS.REACT_LOAD_ROOM, {
    room: nextRoom,
    spawnDoorId: spawnDoor?.id,
    completedNPCs: gameState.state.completedNPCs,
    completedZones: gameState.state.completedZones,
    collectedItems: gameState.state.collectedItems,
    doorStates,
  });
}, [gameState, rooms]);
```

**BootScene → ExplorationScene startup:**
```typescript
useEffect(() => {
  const handleSceneReady = (sceneKey: string) => {
    if (sceneKey === 'Boot' && gameRef.current) {
      const entranceRoom = rooms.find(r => r.id === 'hospital_entrance');
      if (!entranceRoom) return;
      const completedRooms = gameState.state.completedRooms;
      const resumeRoomId = gameState.state.currentRoomId ?? 'hospital_entrance';
      const startRoom = rooms.find(r => r.id === resumeRoomId) ?? entranceRoom;
      const doorStates = computeDoorStates(startRoom, completedRooms);
      gameRef.current.scene.start('Exploration', {
        room: startRoom,
        completedNPCs: gameState.state.completedNPCs,
        completedZones: gameState.state.completedZones,
        collectedItems: gameState.state.collectedItems,
        doorStates,
      });
    }
  };
  eventBridge.on(BRIDGE_EVENTS.SCENE_READY, handleSceneReady);
  return () => eventBridge.off(BRIDGE_EVENTS.SCENE_READY, handleSceneReady);
}, [gameState.state.currentRoomId]);
```
