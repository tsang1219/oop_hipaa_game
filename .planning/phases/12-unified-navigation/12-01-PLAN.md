**Unlock logic (export as standalone functions for testability):**
```typescript
export const UNLOCK_ORDER = [
  'hospital_entrance', 'reception', 'break_room', 'lab', 'records_room', 'it_office', 'er'
];

export function isDepartmentUnlocked(roomId: string, completedRooms: string[]): boolean {
  const idx = UNLOCK_ORDER.indexOf(roomId);
  if (idx <= 0) return true; // entrance + anything not in the list = always open
  const prerequisite = UNLOCK_ORDER[idx - 1];
  return completedRooms.includes(prerequisite);
}

export function isDepartmentAccessible(roomId: string, completedRooms: string[]): boolean {
  if (completedRooms.includes(roomId)) return true; // backtrack always allowed
  return isDepartmentUnlocked(roomId, completedRooms);
}
```

Hallway rooms (prefix `hallway_`) are accessible when their preceding department is complete or accessible — treat them as unlocking alongside their preceding department by mapping them to the non-hallway room they follow.
