```typescript
/** Called in encounter mode when all 4 waves are cleared. */
private onEncounterVictory(): void {
  if (!this.encounterId) return;  // safety guard

  const scoreContribution = Math.round((this.securityScore / 100) * 12);

  // Write result to Phaser registry — ExplorationScene reads this on wake
  this.registry.set(`encounterResult_${this.encounterId}`, {
    type: 'td-inbound',
    score: this.securityScore,
    completed: true,
    outcome: 'victory',
    scoreContribution,
  });

  // Emit to React for debrief screen — React updates complianceScore from this
  eventBridge.emit(BRIDGE_EVENTS.ENCOUNTER_COMPLETE, {
    encounterId: this.encounterId,
    outcome: 'victory',
    securityScore: this.securityScore,
    scoreContribution,
  });

  // After a brief delay (let the victory text play), fade out and wake ExplorationScene
  this.time.delayedCall(1800, () => {
    this.cameras.main.fadeOut(400, 0, 0, 0);
    this.cameras.main.once(
      Phaser.Cameras.Scene2D.Events.FADE_OUT_COMPLETE,
      () => {
        this.scene.stop();                   // full shutdown, frees memory
        this.scene.wake('Exploration');      // resumes RPG at exact player position
      }
    );
  });
}

/** Called in encounter mode when securityScore reaches 0 (game over). */
private onEncounterGameOver(): void {
  if (!this.encounterId) return;  // safety guard

  // Even on defeat, award partial score contribution for partial defense
  const scoreContribution = Math.round((this.securityScore / 100) * 12);

  this.registry.set(`encounterResult_${this.encounterId}`, {
    type: 'td-inbound',
    score: this.securityScore,
    completed: true,
    outcome: 'defeat',
    scoreContribution,
  });

  eventBridge.emit(BRIDGE_EVENTS.ENCOUNTER_COMPLETE, {
    encounterId: this.encounterId,
    outcome: 'defeat',
    securityScore: this.securityScore,
    scoreContribution,
  });

  this.time.delayedCall(2000, () => {
    this.cameras.main.fadeOut(400, 0, 0, 0);
    this.cameras.main.once(
      Phaser.Cameras.Scene2D.Events.FADE_OUT_COMPLETE,
      () => {
        this.scene.stop();
        this.scene.wake('Exploration');
      }
    );
  });
}
```
